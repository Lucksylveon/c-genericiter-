CC      = gcc
CFLAGS  = -Wall -Wextra -Wpedantic -std=c11
TARGET  = game_v2
SRCS    = main_v2.c entity_v2.c mage_v2.c warrior_v2.c
OBJS    = $(SRCS:.c=.o)

all: $(TARGET)

$(TARGET): $(OBJS)
	$(CC) $(CFLAGS) -o $@ $^
	@echo "✅ Compilation réussie → ./$(TARGET)"

%.o: %.c
	$(CC) $(CFLAGS) -c $< -o $@

debug: CFLAGS += -DDEBUG -g3 -fsanitize=address
debug: $(TARGET)

run: all
	./$(TARGET)

clean:
	rm -f $(OBJS) $(TARGET)

.PHONY: all debug run clean
